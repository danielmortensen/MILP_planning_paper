			\addplot [fill=green!20, bar width=0.15] table [x expr=\coordindex*0.6 + 0.3, y=algorithm1]{\table};
			\addlegendentry{He et al.}

			\addlegendentry{Optimized}

		\end{axis}
	\end{tikzpicture}
